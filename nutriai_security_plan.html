HTTPS-only enforcement + HSTS Critical
All traffic must be encrypted in transit. Render provides HTTPS automatically but HSTS forces browsers to always use HTTPS even if a user types http://. Add to every response header so browsers cache the HTTPS-only rule for 1 year.
@app.after_request def add_security_headers(response): response.headers['Strict-Transport-Security'] = 'max-age=31536000; includeSubDomains' response.headers['X-Content-Type-Options'] = 'nosniff' response.headers['X-Frame-Options'] = 'DENY' response.headers['X-XSS-Protection'] = '1; mode=block' response.headers['Referrer-Policy'] = 'strict-origin-when-cross-origin' response.headers['Content-Security-Policy'] = ( "default-src 'self'; " "script-src 'self' 'unsafe-inline' https://cdn.jsdelivr.net https://cdnjs.cloudflare.com; " "style-src 'self' 'unsafe-inline' https://fonts.googleapis.com; " "font-src 'self' https://fonts.gstatic.com; " "img-src 'self' data: https:; " "connect-src 'self';" ) return response

Session security hardening High
Flask sessions are signed cookies. Without proper config, they can be hijacked via XSS or stolen over HTTP. These 4 settings make sessions cryptographically secure and scope them to the current request context.
app.config.update( SECRET_KEY = os.environ['SECRET_KEY'], # min 64 random hex chars SESSION_COOKIE_SECURE = True, # only send over HTTPS SESSION_COOKIE_HTTPONLY = True, # JS cannot access cookie SESSION_COOKIE_SAMESITE = 'Lax', # CSRF protection PERMANENT_SESSION_LIFETIME = 86400, # 24-hour session lifetime SESSION_COOKIE_NAME = '_nutriai_session', )
🛡️
CSRF protection on all state-changing forms High
Cross-Site Request Forgery attacks trick a logged-in user's browser into submitting forms to your site from another site. Every POST form (register, login, forgot-password, analyze, diary add) needs a CSRF token. Flask-WTF adds this automatically. Without it, an attacker can forge requests on behalf of your users.
pip install flask-wtf # In config.py WTF_CSRF_ENABLED = True WTF_CSRF_SECRET_KEY = os.environ.get('SECRET_KEY') # In app.py from flask_wtf.csrf import CSRFProtect csrf = CSRFProtect(app) # In every HTML form <form method="POST"> <input type="hidden" name="csrf_token" value="{{ csrf_token() }}"> ... </form> # AJAX requests need the token in headers const resp = await fetch('/api/diary/add', { method: 'POST', headers: {'X-CSRFToken': document.cookie.match(/csrf_token=([^;]+)/)?.[1]}, body: JSON.stringify(data) })

🔵 API & input security
📏
Strict server-side input validation on all fields Medium
Browser-side validation (HTML min/max, required) can be bypassed with curl or Burp Suite in seconds. Every field must be validated on the server before touching the database or AI. The age clamping fix is already done — extend this pattern to all fields.
def validate_analyze_input(form): errors = [] # Numeric bounds age = min(max(int(form.get('age', 25)), 10), 100) height = min(max(float(form.get('height', 170)), 50), 300) weight = min(max(float(form.get('weight', 70)), 10), 500) # Enum validation — only allow known values allowed_goals = {'weight_loss','muscle_gain','maintain'} allowed_diets = {'vegetarian','vegan','non_vegetarian'} allowed_genders = {'male','female','other'} goal = form.get('goal','maintain') diet = form.get('diet_type','non_vegetarian') gender = form.get('gender','male') if goal not in allowed_goals: errors.append('Invalid goal') if diet not in allowed_diets: errors.append('Invalid diet type') if gender not in allowed_genders: errors.append('Invalid gender') # Text field sanitization (strip HTML tags) import re name = re.sub(r'<[^>]+>', '', form.get('name','User'))[:100] if errors: raise ValueError('; '.join(errors)) return age, height, weight, goal, diet, gender, name
🖼️
File upload security for food image scan Medium
The /api/analyze-food-image endpoint accepts base64-encoded images. Without validation, an attacker could send malformed data, extremely large payloads, or non-image files. All three must be caught before the data reaches Gemini Vision.
ALLOWED_MIME_TYPES = {'image/jpeg', 'image/png', 'image/webp', 'image/gif'} MAX_IMAGE_SIZE_BYTES = 10 * 1024 * 1024 # 10MB def validate_image_upload(image_data: str, mime_type: str): # 1. Check MIME type whitelist if mime_type not in ALLOWED_MIME_TYPES: raise ValueError(f'Unsupported image type: {mime_type}') # 2. Strip data URI prefix if present if ',' in image_data: image_data = image_data.split(',', 1)[1] # 3. Validate base64 encoding import base64, binascii try: raw = base64.b64decode(image_data) except (binascii.Error, ValueError): raise ValueError('Invalid base64 encoding') # 4. Check decoded size if len(raw) > MAX_IMAGE_SIZE_BYTES: raise ValueError('Image exceeds 10MB limit') # 5. Check magic bytes (file signature) MAGIC = { b'\xff\xd8\xff': 'image/jpeg', b'\x89PNG': 'image/png', b'RIFF': 'image/webp', b'GIF8': 'image/gif', } matched = any(raw.startswith(sig) for sig in MAGIC) if not matched: raise ValueError('File content does not match declared type') return raw

📊 Logging — see everything that happens
📝
Structured security event logging Critical
Without logging, breaches are invisible. Add structured JSON logging to every security-relevant event: failed logins, rate limit hits, blocked requests, admin actions, password resets, account deletions. Render streams these to its log dashboard and you can grep them.
import logging, json from datetime import datetime logger = logging.getLogger('nutriai.security') logging.basicConfig(level=logging.INFO, format='%(message)s') def sec_log(event: str, ip: str, extra: dict = None): """Emit a structured security event to stdout (captured by Render logs).""" payload = { 'ts': datetime.utcnow().isoformat(), 'event': event, 'ip': ip, **(extra or {}) } logger.warning(json.dumps(payload)) # Usage throughout app.py: # Failed login sec_log('LOGIN_FAIL', ip, {'email': email, 'reason': err}) # Rate limit hit sec_log('RATE_LIMIT', ip, {'endpoint': request.path, 'limit': '5/10min'}) # Admin login sec_log('ADMIN_LOGIN', ip, {'username': username, 'success': True}) # Suspicious input (XSS attempt detected) sec_log('SUSPICIOUS_INPUT', ip, {'field': 'name', 'value': name[:50]}) # Password reset requested sec_log('PASSWORD_RESET_REQ', ip, {'email': email})
🚦
Failed authentication tracking + auto-lockout High
The rate limiter blocks by IP but a distributed attack from many IPs (botnet) would bypass it. Track consecutive failed login attempts per email address and temporarily lock the account after 10 failures within 15 minutes, regardless of source IP.
_failed_logins: dict = {} # {email: [timestamp, ...]} def record_failed_login(email: str) -> bool: """Returns True if account should be locked.""" import time now = time.time() window = 900 # 15 minutes # Keep only attempts within window attempts = [t for t in _failed_logins.get(email, []) if now - t < window] attempts.append(now) _failed_logins[email] = attempts # Lock after 10 failures in window return len(attempts) >= 10 def is_account_locked(email: str) -> bool: import time now = time.time() attempts = [t for t in _failed_logins.get(email, []) if now - t < 900] return len(attempts) >= 10 # In login route: if is_account_locked(email): sec_log('ACCOUNT_LOCKED', ip, {'email': email}) return render_template('auth/login.html', error='Account temporarily locked. Try again in 15 minutes.'), 429 acc, err = login_account(email, password) if err: if record_failed_login(email): sec_log('ACCOUNT_LOCKED', ip, {'email': email, 'attempts': 10}) else: sec_log('LOGIN_FAIL', ip, {'email': email})

Anomaly detection on API usage patterns High
Detect unusual usage patterns that suggest abuse: same IP generating plans 24/7, voice log being called with very long transcripts (prompt injection attempt), unusually large request payloads, unusual User-Agent strings.
def detect_anomalies(request): ip = get_ip(request) ua = request.headers.get('User-Agent', '') size = request.content_length or 0 anomalies = [] # Suspicious User-Agents (bots, scanners, scrapers) suspicious_ua = ['sqlmap', 'nikto', 'nmap', 'masscan', 'zgrab', 'python-requests', 'curl/', 'wget/', 'scrapy'] if any(s in ua.lower() for s in suspicious_ua): anomalies.append(f'suspicious_ua:{ua[:50]}') # Unusually large request (potential DoS or injection) if size > 1024 * 1024: # 1MB anomalies.append(f'large_request:{size}') # Missing User-Agent (bot behaviour) if not ua: anomalies.append('missing_ua') # Voice transcript over 500 chars (prompt injection attempt) if request.path == '/api/voice-log': data = request.get_json(silent=True) or {} if len(data.get('transcript', '')) > 500: anomalies.append('long_transcript') if anomalies: sec_log('ANOMALY', ip, {'path': request.path, 'flags': anomalies}) return anomalies
